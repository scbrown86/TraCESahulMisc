library(TraCESahulMisc)
library(terra)

fn <- c("D:/TraCE-Sahul/pr/TraCE_22ka_downscaled_pr_decadal_21k_1500CE_biascorr_06.nc",
        "D:/TraCE-Sahul/pr/TraCE_22ka_downscaled_pr_1500_1990_biascorr.nc")

# import the data
sahul_pr <- import_TraCESahul(fn)
sahul_pr

# summarise the data to annual and seasonal timesteps
pr_ann <- summarise_TraCESahul(x = sahul_pr, type = "annual",
                               sumfun = "mean", window = NULL)
pr_ann
pr_season <- summarise_TraCESahul(x = sahul_pr, type = "seasonal",
                                  sumfun = "mean", window = NULL)
pr_season

pr_monthly <- summarise_TraCESahul(x = sahul_pr, type = "monthly",
                                   sumfun = "mean", window = NULL)
pr_monthly

pr_ann_win <- summarise_TraCESahul(x = sahul_pr, type = "annual",
                                   sumfun = "mean", window = 30)
pr_ann_win

pr_season_win <- summarise_TraCESahul(x = sahul_pr, type = "seasonal",
                                      sumfun = "mean", window = 30)
pr_season_win

pr_monthly_win <- summarise_TraCESahul(x = sahul_pr, type = "monthly",
                                       sumfun = "mean", window = 30)
pr_monthly_win
